  5 NAV2D.OccupancyGridClientNav = function(options) {
  6   var that = this;
  7   var options = options || {};
  8   this.ros = options.ros;
  9   var topic = options.topic || '/map';
 10   var continuous = options.continuous;
 11   this.serverName = options.serverName || '/move_base';
 12   this.actionName = options.actionName || 'move_base_msgs/MoveBaseAction';
 13   this.rootObject = options.rootObject || new createjs.Container();
 14   this.viewer = options.viewer;
 15 
 16   this.navigator = null;
 17 
 18   // setup a client to get the map
 19   var client = new ROS2D.OccupancyGridClient({
 20     ros : this.ros,
 21     rootObject : this.rootObject,
 22     continuous : continuous
 23   });
 24   client.on('change', function() {
 25     that.navigator = new NAV2D.Navigator({
 26       ros : that.ros,
 27       serverName : that.serverName,
 28       actionName : that.actionName,
 29       rootObject : that.rootObject
 30     });
 31     
 32     // scale the viewer to fit the map
 33     that.viewer.scaleToDimensions(client.currentGrid.width, client.currentGrid.height);
 34   });
 35 };
