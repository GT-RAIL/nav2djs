  5 NAV2D.Navigator = function(options) {
  6   var that = this;
  7   var options = options || {};
  8   var ros = options.ros;
  9   var serverName = options.serverName || '/move_base';
 10   var actionName = options.actionName || 'move_base_msgs/MoveBaseAction';
 11   this.rootObject = options.rootObject || new createjs.Container();
 12 
 13   // setup the actionlib client
 14   var actionClient = new ROSLIB.ActionClient({
 15     ros : ros,
 16     actionName : actionName,
 17     serverName : serverName
 18   });
 19 
 20   /**
 21    * Send a goal to the navigation stack with the given pose.
 22    * 
 23    * @param pose - the goal pose
 24    */
 25   var sendGoal = function(pose) {
 26     // create a goal
 27     var goal = new ROSLIB.Goal({
 28       actionClient : actionClient,
 29       goalMessage : {
 30         target_pose : {
 31           header : {
 32             frame_id : '/map'
 33           },
 34           pose : pose
 35         }
 36       }
 37     });
 38     goal.send();
 39 
 40     // create a marker for the goal
 41     var goalMarker = new ROS2D.NavigationArrow({
 42       size : 8,
 43       strokeSize : 1,
 44       fillColor : createjs.Graphics.getRGB(255, 64, 128, 0.66)
 45     });
 46     goalMarker.x = pose.position.x;
 47     goalMarker.y = -pose.position.y;
 48     goalMarker.rotation = stage.rosQuaternionToGlobalTheta(pose.orientation);
 49     goalMarker.scaleX = 1.0 / stage.scaleX;
 50     goalMarker.scaleY = 1.0 / stage.scaleY;
 51     that.rootObject.addChild(goalMarker);
 52 
 53     // have the model "pulse"
 54     var growCount = 0;
 55     var growing = true;
 56     createjs.Ticker.addEventListener('tick', function() {
 57       if (growing) {
 58         goalMarker.scaleX *= 1.035;
 59         goalMarker.scaleY *= 1.035;
 60         growing = (++growCount < 10);
 61       } else {
 62         goalMarker.scaleX /= 1.035;
 63         goalMarker.scaleY /= 1.035;
 64         growing = (--growCount < 0);
 65       }
 66     });
 67     
 68     goal.on('result', function() {
 69       that.rootObject.removeChild(goalMarker);
 70     });
 71   };
 72 
 73   // get a handle to the stage
 74   if (that.rootObject instanceof createjs.Stage) {
 75     var stage = that.rootObject;
 76   } else {
 77     var stage = that.rootObject.getStage();
 78   }
 79 
 80   // marker for the robot
 81   var robotMarker = new ROS2D.NavigationArrow({
 82     size : 12,
 83     strokeSize : 1,
 84     fillColor : createjs.Graphics.getRGB(255, 128, 0, 0.66)
 85   });
 86   // wait for a pose to come in first
 87   robotMarker.visible = false;
 88   this.rootObject.addChild(robotMarker);
 89   var initScaleSet = false;
 90   // have the model "pulse"
 91   var growCount = 0;
 92   var growing = true;
 93   createjs.Ticker.addEventListener('tick', function() {
 94     if (initScaleSet) {
 95       if (growing) {
 96         robotMarker.scaleX *= 1.035;
 97         robotMarker.scaleY *= 1.035;
 98         growing = (++growCount < 10);
 99       } else {
100         robotMarker.scaleX /= 1.035;
101         robotMarker.scaleY /= 1.035;
102         growing = (--growCount < 0);
103       }
104     }
105   });
106 
107   // setup a listener for the robot pose
108   var poseListener = new ROSLIB.Topic({
109     ros : ros,
110     name : '/robot_pose',
111     messageType : 'geometry_msgs/Pose',
112     throttle_rate : 100
113   });
114   poseListener.subscribe(function(pose) {
115     // update the robots position on the map
116     robotMarker.x = pose.position.x;
117     robotMarker.y = -pose.position.y;
118     if (!initScaleSet) {
119       robotMarker.scaleX = 1.0 / stage.scaleX;
120       robotMarker.scaleY = 1.0 / stage.scaleY;
121       initScaleSet = true;
122     }
123 
124     // change the angle
125     robotMarker.rotation = stage.rosQuaternionToGlobalTheta(pose.orientation);
126 
127     robotMarker.visible = true;
128   });
129 
130   // setup a double click listener (no orientation)
131   this.rootObject.addEventListener('dblclick', function(event) {
132     // convert to ROS coordinates
133     var coords = stage.globalToRos(event.stageX, event.stageY);
134     var pose = new ROSLIB.Pose({
135       position : new ROSLIB.Vector3(coords)
136     });
137     // send the goal
138     sendGoal(pose);
139   });
140 };
